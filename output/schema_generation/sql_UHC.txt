CREATE TABLE `patients` ("patient_id" INT, "first_name" VARCHAR, "last_name" VARCHAR, "date_of_birth" DATE, "age" INT, "gender" VARCHAR, "address" VARCHAR, "phone_number" CHAR, "email" VARCHAR, "emergency_contact_name" VARCHAR, "emergency_contact_phone" CHAR, "insurance_provider" VARCHAR, "insurance_policy_number" VARCHAR, "registration_date" DATE, "active_status" BOOLEAN, "medical_history" TEXT, "allergies" VARCHAR, "medications" VARCHAR, "preferred_language" VARCHAR, "ethnicity" VARCHAR, "marital_status" VARCHAR, PRIMARY KEY ("patient_id"))
CREATE TABLE `doctors` ("doctor_id" INT, "first_name" VARCHAR, "last_name" VARCHAR, "date_of_birth" DATE, "gender" VARCHAR, "specialization" VARCHAR, "address" VARCHAR, "phone_number" CHAR, "email" VARCHAR, "license_number" VARCHAR, "hire_date" DATE, "active_status" BOOLEAN, "qualifications" TEXT, "languages_spoken" VARCHAR, "department" VARCHAR, PRIMARY KEY ("doctor_id"))
CREATE TABLE `appointments` ("appointment_id" INT, "appointment_date" DATE, "appointment_time" TIME, "appointment_type" VARCHAR, "notes" TEXT, "completed" BOOLEAN, PRIMARY KEY ("appointment_id"))
CREATE TABLE `medical_records` ("record_id" INT, "record_date" DATE, "diagnosis" TEXT, "treatment" TEXT, "prescriptions" TEXT, "notes" TEXT, PRIMARY KEY ("record_id"))
CREATE TABLE `billing` ("billing_id" INT, "amount_due" REAL, "amount_paid" REAL, "billing_date" DATE, "payment_method" VARCHAR, "paid_in_full" BOOLEAN, PRIMARY KEY ("billing_id"))
