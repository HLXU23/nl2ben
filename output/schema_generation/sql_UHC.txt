CREATE TABLE `patients` ("patient_id" INT, "first_name" VARCHAR, "last_name" VARCHAR, "date_of_birth" DATE, "age" INT, "gender" VARCHAR, "address" VARCHAR, "phone_number" CHAR, "email" VARCHAR, "emergency_contact" VARCHAR, "emergency_phone" CHAR, "insurance_provider" VARCHAR, "insurance_id" VARCHAR, "registration_date" DATE, "active_status" BOOLEAN, "medical_history" TEXT, "allergies" VARCHAR, "medications" VARCHAR, "blood_type" VARCHAR, "preferred_language" VARCHAR, "ethnicity" VARCHAR, PRIMARY KEY ("patient_id"))
CREATE TABLE `doctors` ("doctor_id" INT, "first_name" VARCHAR, "last_name" VARCHAR, "date_of_birth" DATE, "gender" VARCHAR, "specialization" VARCHAR, "address" VARCHAR, "phone_number" CHAR, "email" VARCHAR, "license_number" VARCHAR, "hire_date" DATE, "active_status" BOOLEAN, "biography" TEXT, "languages" VARCHAR, PRIMARY KEY ("doctor_id"))
CREATE TABLE `appointments` ("appointment_id" INT, "appointment_datetime" TIMESTAMP, "appointment_type" VARCHAR, "notes" TEXT, "confirmed" BOOLEAN, PRIMARY KEY ("appointment_id"))
CREATE TABLE `medical_records` ("record_id" INT, "record_date" DATE, "diagnosis" TEXT, "treatment" TEXT, "prescriptions" TEXT, "test_results" TEXT, PRIMARY KEY ("record_id"))
CREATE TABLE `billing` ("billing_id" INT, "amount" REAL, "billing_date" DATE, "payment_method" VARCHAR, "paid" BOOLEAN, "description" TEXT, PRIMARY KEY ("billing_id"))
